### ✅ 题目 1：反转字符串
**难度**: ⭐ | **考察点**: 双指针

```javascript
// 把字符串反转过来
function reverseString(s) {
    let left = 0;
    let right = s.length - 1;
    
    while(left < right) {
        // 交换左右字符
        [s[left], s[right]] = [s[right], s[left]];
        left++;
        right--;
    }
    
    return s;
}

// 测试
console.log(reverseString(["h","e","l","l","o"])); 
// ["o","l","l","e","h"]
```

**关键点**：
- 双指针，一个从头，一个从尾
- 相遇时停止

### ✅ 题目 2：数组去重
**难度**: ⭐⭐ | **考察点**: 哈希表 / 双指针

#### 方法 1：使用 Set
```javascript
function removeDuplicates(nums) {
    return [...new Set(nums)];
}

// 测试
console.log(removeDuplicates([1,1,2,2,3])); // [1,2,3]
```

#### 方法 2：双指针（原地修改）
```javascript
function removeDuplicates(nums) {
    if(nums.length === 0) return 0;
    
    let i = 0;  // 慢指针
    for(let j = 1; j < nums.length; j++) {  // 快指针
        if(nums[j] !== nums[i]) {
            i++;
            nums[i] = nums[j];
        }
    }
    
    return i + 1;  // 返回新长度
}

// 测试
let arr = [1, 1, 2, 2, 3];
let len = removeDuplicates(arr);
console.log(arr.slice(0, len)); // [1,2,3]
```

### ✅ 题目 3：最大子数组和
**难度**: ⭐⭐ | **考察点**: 动态规划

```javascript
// 找到连续子数组的最大和
function maxSubArray(nums) {
    let maxSum = nums[0];  // 记录全局最大值
    let currentSum = nums[0];  // 以当前位置结尾的最大和
    
    for(let i = 1; i < nums.length; i++) {
        // 要么延续之前的和，要么从当前重新开始
        currentSum = Math.max(currentSum + nums[i], nums[i]);
        // 更新全局最大值
        maxSum = Math.max(maxSum, currentSum);
    }
    
    return maxSum;
}

// 测试
console.log(maxSubArray([-2,1,-3,4,-1,2,1,-5,4])); // 6 (子数组 [4,-1,2,1])
console.log(maxSubArray([1])); // 1
console.log(maxSubArray([5,4,-1,7,8])); // 23
```

**核心思想**：
- 如果前面的和是负数，就不要了，从当前重新开始
- 每次都记录历史最大值

### ✅ 题目 4：合并两个有序数组
**难度**: ⭐⭐ | **考察点**: 双指针

```javascript
// 合并两个有序数组，结果存在 nums1 中
function merge(nums1, m, nums2, n) {
    let i = m - 1;  // nums1 的末尾（有效元素）
    let j = n - 1;  // nums2 的末尾
    let k = m + n - 1;  // 合并后的末尾
    
    // 从后往前放，避免覆盖 nums1 的元素
    while(j >= 0) {
        if(i >= 0 && nums1[i] > nums2[j]) {
            nums1[k] = nums1[i];
            i--;
        } else {
            nums1[k] = nums2[j];
            j--;
        }
        k--;
    }
    
    return nums1;
}

// 测试
let arr1 = [1,2,3,0,0,0];
merge(arr1, 3, [2,5,6], 3);
console.log(arr1); // [1,2,2,3,5,6]
```

**关键点**：
- 从后往前放！从后往前放！从后往前放！
- 避免覆盖 nums1 前面的元素

### 🔥 题目 7：反转链表

```javascript
// 反转单链表
function reverseList(head) {
    let prev = null;
    let curr = head;
    
    while(curr) {
        let nextTemp = curr.next;  // 先保存下一个
        curr.next = prev;          // 反转
        prev = curr;               // 前进
        curr = nextTemp;           // 前进
    }
    
    return prev;
}
```

**记忆口诀**：
> 一保存（保存 next）
> 二反转（curr.next = prev）
> 三前进（prev 和 curr 都向前）

### 🔥 题目 8：环形链表检测
**难度**: ⭐⭐⭐ | **考察点**: 快慢指针

```javascript
// 判断链表是否有环
function hasCycle(head) {
    if(!head || !head.next) return false;
    
    let slow = head;
    let fast = head.next;
    
    while(fast && fast.next) {
        if(slow === fast) return true;  // 相遇说明有环
        
        slow = slow.next;           // 慢的走一步
        fast = fast.next.next;      // 快的走两步
    }
    
    return false;
}
```

**生活比喻**：
- 操场跑步，快的会套圈慢的
- 如果有环，快慢指针一定会相遇

### 🔥 题目 9：二叉树的最大深度
**难度**: ⭐⭐⭐ | **考察点**: 递归

```javascript
// 计算二叉树的最大深度
function maxDepth(root) {
    if(!root) return 0;  // 空树深度为 0
    
    // 递归计算左右子树的深度
    let leftDepth = maxDepth(root.left);
    let rightDepth = maxDepth(root.right);
    
    // 取较大的那个 + 1（加上根节点）
    return Math.max(leftDepth, rightDepth) + 1;
}
```

**理解方式**：
- 树的深度 = max(左子树深度，右子树深度) + 1
- 递归终止条件：空节点返回 0

### 🔥 题目 10：对称二叉树
**难度**: ⭐⭐⭐ | **考察点**: 递归

```javascript
// 判断二叉树是否对称
function isSymmetric(root) {
    if(!root) return true;
    
    return isMirror(root.left, root.right);
}

function isMirror(t1, t2) {
    // 都为空，对称
    if(!t1 && !t2) return true;
    
    // 只有一个为空，不对称
    if(!t1 || !t2) return false;
    
    // 值不相等，不对称
    if(t1.val !== t2.val) return false;
    
    // 递归检查：左的左 vs 右的右，左的右 vs 右的左
    return isMirror(t1.left, t2.right) && 
           isMirror(t1.right, t2.left);
}
```

### 💎 题目 12：接雨水
**难度**: ⭐⭐⭐⭐⭐ | **考察点**: 双指针 / 动态规划

```javascript
// 计算能接多少雨水
function trap(height) {
    if(!height || height.length === 0) return 0;
    
    let left = 0;
    let right = height.length - 1;
    let leftMax = 0;
    let rightMax = 0;
    let water = 0;
    
    while(left < right) {
        if(height[left] < height[right]) {
            if(height[left] >= leftMax) {
                leftMax = height[left];
            } else {
                water += leftMax - height[left];
            }
            left++;
        } else {
            if(height[right] >= rightMax) {
                rightMax = height[right];
            } else {
                water += rightMax - height[right];
            }
            right--;
        }
    }
    
    return water;
}

// 测试
console.log(trap([0,1,0,2,1,0,1,3,2,1,2,1])); // 6
```

**核心思想**：
- 每个位置能接的水 = min(左边最高，右边最高) - 当前高度
- 用双指针从两边向中间移动
