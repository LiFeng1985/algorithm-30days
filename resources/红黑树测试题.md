### **题目 2：验证 BST 性质**

```javascript
// 添加一个方法来验证是否是合法的二叉搜索树

function isValidBST(tree) {
    // 提示：中序遍历应该有序
    
    function validate(node, min, max) {
        if (!node) return true;
        
        if (node.data <= min || node.data >= max) {
            return false;
        }
        
        return validate(node.left, min, node.data) && 
               validate(node.right, node.data, max);
    }
    
    return validate(tree.root, -Infinity, Infinity);
}

// 测试
const tree = new RedBlackTree();
[40, 20, 60, 10, 30, 50, 70].forEach(x => tree.insert(x));
console.log(isValidBST(tree));  // 应该输出 true
```
