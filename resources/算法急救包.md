## 一、归并排序（Merge Sort）

### 生活例子
想象你在整理两摞已经排好序的扑克牌：
- 左边一摞：2, 5, 9
- 右边一摞：1, 5, 6

每次从两摞的最上面选**最小的**那张，放到新堆里。

### 代码模板（背下来！）

```javascript
// 主函数：负责分解
function mergeSort(arr) {
    if(arr.length <= 1) return arr;  // 只有一个元素，返回
    
    let mid = Math.floor(arr.length / 2);  // 找中点
    let left = mergeSort(arr.slice(0, mid));   // 递归左半
    let right = mergeSort(arr.slice(mid));     // 递归右半
    return merge(left, right);  // 合并
}

// 辅助函数：负责合并（关键！）
function merge(left, right) {
    let result = [];
    let i = 0, j = 0;
    
    // 两个指针比赛
    while(i < left.length && j < right.length) {
        if(left[i] < right[j]) {
            result.push(left[i]);
            i++;
        } else {
            result.push(right[j]);
            j++;
        }
    }
    
    // 把剩下的加到末尾
    return result.concat(left.slice(i)).concat(right.slice(j));
}

// 测试
console.log(mergeSort([5, 2, 9, 1, 5, 6]));
// [1, 2, 5, 5, 6, 9]
```

### 记忆口诀
> 一分二，二分四，直到只剩一个
> 左右比，小的走，最后拼剩余

## 二、堆排序（Heap Sort）

### 生活例子
想象一群小朋友按身高排队，每次让最高的出列：
1. 先让他们站成大顶堆（最高的在顶端）
2. 把最高的和最后一个交换
3. 重新调整，再让次高的出来
4. 重复...

### 完全二叉树的数组表示
```
索引关系：
父节点：i
左孩子：2*i + 1
右孩子：2*i + 2
```

### 代码模板（背下来！）

```javascript
function heapSort(arr) {
    let n = arr.length;
    
    // 第一步：建堆（从最后一个非叶子节点开始）
    for(let i = Math.floor(n/2) - 1; i >= 0; i--) {
        heapify(arr, n, i);
    }
    
    // 第二步：不断把堆顶换到末尾
    for(let i = n - 1; i > 0; i--) {
        [arr[0], arr[i]] = [arr[i], arr[0]];  // 交换
        heapify(arr, i, 0);  // 重新调整
    }
    return arr;
}

// heapify：维护大顶堆
function heapify(arr, n, i) {
    let largest = i;  // 假设当前最大
    let left = 2 * i + 1;   // 左孩子
    let right = 2 * i + 2;  // 右孩子
    
    // 找最大的
    if(left < n && arr[left] > arr[largest]) {
        largest = left;
    }
    if(right < n && arr[right] > arr[largest]) {
        largest = right;
    }
    
    // 如果最大的不是父节点，交换并继续调整
    if(largest !== i) {
        [arr[i], arr[largest]] = [arr[largest], arr[i]];
        heapify(arr, n, largest);
    }
}

// 测试
console.log(heapSort([5, 2, 9, 1, 5, 6]));
// [1, 2, 5, 5, 6, 9]
```

### 记忆口诀
> 建堆从后往前扫（非叶节点开始）
> 调整比较父子仨（父、左、右）
> 交换堆顶到末尾
> 递归调整不能忘

## 三、栈（Stack）

### 特性
**LIFO**：后进先出（Last In, First Out）

### 生活例子
- 一摞盘子：最后放上去的，最先被拿走
- 浏览器后退：最后访问的页面，先退回去

### 代码模板

```javascript
class Stack {
    constructor() {
        this.items = [];
    }
    
    push(item) {
        this.items.push(item);  // 压入栈顶
    }
    
    pop() {
        if(this.isEmpty()) return null;
        return this.items.pop();  // 弹出栈顶
    }
    
    peek() {
        if(this.isEmpty()) return null;
        return this.items[this.items.length - 1];  // 查看栈顶
    }
    
    isEmpty() {
        return this.items.length === 0;
    }
}
```

## 四、队列（Queue）

### 特性
**FIFO**：先进先出（First In, First Out）

### 生活例子
- 排队买票：先来的人先买到
- 打印机任务：先提交的任务先打印

### 代码模板

```javascript
class Queue {
    constructor() {
        this.items = [];
    }
    
    enqueue(item) {
        this.items.unshift(item);  // 从队尾加入
    }
    
    dequeue() {
        if(this.isEmpty()) return null;
        return this.items.shift();  // 从队头移除
    }
    
    front() {
        if(this.isEmpty()) return null;
        return this.items[0];  // 查看队头
    }
    
    isEmpty() {
        return this.items.length === 0;
    }
}
```

## 五、链表（Linked List）

### 结构
```
Node {
    data: 数据
    next: 指向下一个节点
}
```

### 代码模板

```javascript
class Node {
    constructor(data) {
        this.data = data;
        this.next = null;
    }
}

class LinkedList {
    constructor() {
        this.head = null;
    }
    
    // 头部插入
    insertAtHead(data) {
        let node = new Node(data);
        node.next = this.head;
        this.head = node;
    }
    
    // 尾部插入
    insertAtTail(data) {
        let node = new Node(data);
        if(this.head === null) {
            this.head = node;
        } else {
            let curr = this.head;
            while(curr.next) {
                curr = curr.next;
            }
            curr.next = node;
        }
    }
    
    // 删除节点
    delete(data) {
        if(this.head === null) return;
        
        if(this.head.data === data) {
            this.head = this.head.next;
            return;
        }
        
        let curr = this.head;
        while(curr.next && curr.next.data !== data) {
            curr = curr.next;
        }
        
        if(curr.next) {
            curr.next = curr.next.next;
        }
    }
    
    // 搜索
    search(data) {
        let curr = this.head;
        while(curr) {
            if(curr.data === data) return true;
            curr = curr.next;
        }
        return false;
    }
}
```

## 六、二叉搜索树（BST）

### 性质
- 左子树 < 根节点 < 右子树
- 左右子树也都是 BST

### 代码模板

```javascript
class TreeNode {
    constructor(data) {
        this.data = data;
        this.left = null;
        this.right = null;
    }
}

class BST {
    constructor() {
        this.root = null;
    }
    
    // 搜索
    search(data) {
        let curr = this.root;
        while(curr) {
            if(curr.data === data) return curr;
            if(data < curr.data) {
                curr = curr.left;
            } else {
                curr = curr.right;
            }
        }
        return null;
    }
    
    // 插入（递归）
    insert(root, data) {
        if(root === null) {
            return new TreeNode(data);
        }
        if(data < root.data) {
            root.left = this.insert(root.left, data);
        } else if(data > root.data) {
            root.right = this.insert(root.right, data);
        }
        return root;
    }
    
    // 删除（最难！）
    delete(root, data) {
        if(root === null) return null;
        
        if(data < root.data) {
            root.left = this.delete(root.left, data);
        } else if(data > root.data) {
            root.right = this.delete(root.right, data);
        } else {
            // 找到要删除的节点
            
            // 情况 1：叶子节点
            if(root.left === null && root.right === null) {
                return null;
            }
            
            // 情况 2：只有一个孩子
            if(root.left === null) return root.right;
            if(root.right === null) return root.left;
            
            // 情况 3：有两个孩子
            // 找右子树的最小值替换
            let minNode = this.getMinNode(root.right);
            root.data = minNode.data;
            root.right = this.delete(root.right, minNode.data);
        }
        return root;
    }
    
    // 找最小值（最左边的节点）
    getMinNode(root) {
        while(root.left) {
            root = root.left;
        }
        return root;
    }
}
```

## 七、哈希表（Hash Table）

### 原理
通过哈希函数把 key 映射到数组索引

### 冲突处理
**链地址法**：每个位置是一个链表

### 代码模板

```javascript
class HashMap {
    constructor(size = 16) {
        this.buckets = new Array(size);
    }
    
    _hash(key) {
        // 简单哈希：取模
        return key % this.buckets.length;
    }
    
    put(key, value) {
        let index = this._hash(key);
        
        if(!this.buckets[index]) {
            this.buckets[index] = [];
        }
        
        // 检查是否已存在
        let bucket = this.buckets[index];
        for(let pair of bucket) {
            if(pair[0] === key) {
                pair[1] = value;  // 更新
                return;
            }
        }
        
        bucket.push([key, value]);  // 新增
    }
    
    get(key) {
        let index = this._hash(key);
        let bucket = this.buckets[index];
        
        if(!bucket) return undefined;
        
        for(let pair of bucket) {
            if(pair[0] === key) {
                return pair[1];
            }
        }
        return undefined;
    }
    
    remove(key) {
        let index = this._hash(key);
        let bucket = this.buckets[index];
        
        if(!bucket) return;
        
        for(let i = 0; i < bucket.length; i++) {
            if(bucket[i][0] === key) {
                bucket.splice(i, 1);
                return;
            }
        }
    }
}
```

- 做相关算法题
- 解释给别人听
