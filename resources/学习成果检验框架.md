        let pivotIndex = partition(arr, low, high) // 分区，找到 pivot 位置
        quickSort(arr, low, pivotIndex - 1)   // 递归排序左半部分

function partition(arr, low, high) {
    // TODO: 你的实现
    let pivot = arr[high]              // 选择最后一个元素作为基准
    let i = low - 1                    // i 指向小于 pivot 区域的最后一个元素
    for(let j = low; j < high; j++){   // j 遍历整个数组
        if(arr[j] < pivot){            // 如果当前元素小于 pivot
            i++                        // 扩展小于区域
            [arr[i],arr[j]] = [arr[j],arr[i]]  // 交换到小于区域
        }
    }
    [arr[i+1],arr[high]] = [arr[high],arr[i+1]]  // 将 pivot 放到正确位置
    return i+1    
}

    let n = arr.length

        heapify(arr, n, i)

function heapify(arr, n, i) {
    // TODO: 你的实现
    let largest = i
    let left = i * 2 + 1
    let right = i * 2 + 2
    
    if(left < n && arr[left] > arr[largest]){
        largest = left
    }
    if(right < n && arr[right] > arr[largest]){
        largest = right
    }
    if(largest != i){
        [arr[i],arr[largest]] = [arr[largest],arr[i]]

    }
}

### 🔥 进阶挑战
**难度**: ⭐⭐ | **考察点**: 哈希表

```javascript
// 给定数组和目标值，找出和为目标值的两个数的索引
function twoSum(nums, target) {
    let map = new Map();  // 存储 {数值：索引}
    
    for(let i = 0; i < nums.length; i++){
        let complement = target - nums[i];  // 计算需要的另一个数
        
        if(map.has(complement)) {
            // 找到了，返回两个索引
            return [map.get(complement), i];
        }
        
        // 把当前数和索引存入 map
        map.set(nums[i], i);
    }
    
    return [];  // 没找到
}

// 测试
console.log(twoSum([2, 7, 11, 15], 9)); // [0, 1]
console.log(twoSum([3, 2, 4], 6));      // [1, 2]
console.log(twoSum([3, 3], 6));         // [0, 1]
```

#### 💡 解题思路
**暴力解法**：两层循环，O(n²)
**哈希表解法**：一层循环，O(n) ✅

**关键理解**：
- 遍历数组时，对于每个数 nums[i]，我们需要找 target - nums[i]
- 用哈希表记录已经遍历过的数，这样查找只需要 O(1)

**步骤**：
1. 创建 map，用来存 {数值：索引}
2. 遍历数组
3. 计算 complement = target - nums[i]
4. 检查 map 里有没有 complement
   - 有：返回 [complement 的索引，当前索引]
   - 没有：把当前数存入 map
5. 继续下一个

### 🔥 挑战 2：有效的括号
**难度**: ⭐⭐ | **考察点**: 栈

```javascript
// 使用栈判断括号是否有效
function isValid(s) {
    let stack = [];
    let map = {
        ')': '(',   // 右括号对应左括号
        ']': '[',
        '}': '{'
    };
    
    for(let char of s) {
        // 如果是左括号，入栈
        if(char === '(' || char === '[' || char === '{') {
            stack.push(char);
        } else {
            // 如果是右括号
            // 栈为空说明没有匹配的左括号
            if(stack.length === 0) return false;
            
            // 弹出栈顶，检查是否匹配
            let top = stack.pop();
            if(top !== map[char]) {
                return false;  // 不匹配
            }
        }
    }
    
    // 栈为空说明全部匹配
    return stack.length === 0;
}

// 测试
console.log(isValid("()[]{}"));     // true
console.log(isValid("(]"));         // false
console.log(isValid("([)]"));       // false
console.log(isValid("{[]}"));       // true
console.log(isValid(""));           // true
```

#### 💡 解题思路
**核心思想**：栈（LIFO）

**为什么用栈？**
- 后遇到的左括号要先匹配（就像剥洋葱）
- 例如：`{[]}` 先匹配 `[]`，再匹配 `{}`

**步骤**：
1. 创建映射表：`)`→`(`, `]`→`[`, `}`→`{`
2. 遍历字符串
3. 遇到左括号：压入栈
4. 遇到右括号：
   - 如果栈空了 → false（没有匹配的左括号）
   - 弹出栈顶，检查是否匹配
   - 不匹配 → false
5. 最后检查栈是否为空
   - 空 → true（全部匹配）
   - 非空 → false（有未匹配的左括号）

// 最小堆实现
function mergeKLists(lists) {
    if(!lists || lists.length === 0) return null;
    
    // 最小堆（简化版，实际应该用 Heap 类）
    let heap = [];
    
    // 把所有链表的头节点加入堆
    for(let list of lists) {
        if(list) {
            heap.push(list);
        }
    }
    
    // 按节点值排序（从小到大）
    heap.sort((a, b) => b.val - a.val);
    
    let dummy = new ListNode(0);
    let curr = dummy;
    
    while(heap.length > 0) {
        // 取出最小的节点
        let node = heap.pop();
        curr.next = node;
        curr = curr.next;
        
        // 如果这个节点有下一个，加入堆
        if(node.next) {
            heap.push(node.next);
            heap.sort((a, b) => b.val - a.val);
        }
    }
    
    return dummy.next;
}

#### 方法 2：分治法（推荐）
```javascript
function mergeKLists(lists) {
    if(!lists || lists.length === 0) return null;
    if(lists.length === 1) return lists[0];
    
    // 分成两半
    let mid = Math.floor(lists.length / 2);
    let left = mergeKLists(lists.slice(0, mid));
    let right = mergeKLists(lists.slice(mid));
    
    // 合并两个有序链表
    return mergeTwoLists(left, right);
}

function mergeTwoLists(l1, l2) {
    let dummy = new ListNode(0);
    let curr = dummy;
    
    while(l1 && l2) {
        if(l1.val < l2.val) {
            curr.next = l1;
            l1 = l1.next;
        } else {
            curr.next = l2;
            l2 = l2.next;
        }
        curr = curr.next;
    }
    
    curr.next = l1 ? l1 : l2;
    return dummy.next;
}
```

#### 💡 解题思路
**方法 1：堆**
- 维护一个最小堆，每次取最小的节点
- 时间复杂度：O(N log k)，k 是链表数量

**方法 2：分治法** ✅
- 把 k 个链表两两合并
- 类似归并排序的思想
- 时间复杂度：O(N log k)

**关键理解**：
- 合并 2 个有序链表是最基本的操作
- 分治法：k 个 → k/2 个 → k/4 个 ... → 1 个

## 💡 使用建议

1. **不要急于求成**：每天掌握 2-3 个就很好了
2. **重在理解**：不要死记硬背代码
3. **多动手写**：看懂≠会写
4. **允许犯错**：错误是最好的老师
5. **定期复习**：遗忘是正常的，重复是关键

**记住：编程高手都是练出来的，不是学出来的！** 💪
