- [ ] 插入排序的最好情况时间复杂度是 O(n)

function insertionSortWithLog(arr) {
    const n = arr.length;
    
    for (let i = 1; i < n; i++) {
        console.log(`\n第${i}轮：`);
        console.log('当前数组:', arr);
        console.log('待插入元素:', arr[i]);
        
        const current = arr[i];
        let j = i - 1;

    }
