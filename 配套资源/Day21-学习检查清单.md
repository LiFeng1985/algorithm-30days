### 判断对错

- [ ] 快速排序总是比冒泡排序快
- [ ] 归并排序需要额外空间
- [ ] 堆排序是稳定排序
- [ ] 插入排序适合基本有序的数据

**答案：**
1. ❌（在数据量小或基本有序时，冒泡可能更快）
2. ✅（归并排序需要 O(n) 额外空间）
3. ❌（堆排序是不稳定排序）
4. ✅（插入排序在基本有序时接近 O(n)）

// 性能对比函数
function compareAllSorts() {
    const testCases = [
        { name: '随机数组', data: () => [64, 34, 25, 12, 22, 11, 90] },
        { name: '已有序', data: () => [1, 2, 3, 4, 5, 6, 7] },
        { name: '完全逆序', data: () => [7, 6, 5, 4, 3, 2, 1] },
        { name: '全部相同', data: () => [5, 5, 5, 5, 5] }
    ];
    
    const sorts = [
        { name: '冒泡排序', fn: bubbleSort },
        { name: '选择排序', fn: selectionSort },
        { name: '插入排序', fn: insertionSort },
        { name: '归并排序', fn: mergeSort },
        { name: '快速排序', fn: quickSort },
        { name: '堆排序', fn: heapSort }
    ];
    
    for (const testCase of testCases) {
        console.log(`\n=== ${testCase.name} ===`);
        const originalData = testCase.data();
        
        for (const sort of sorts) {
            const arr = [...originalData];
            const start = performance.now();
            sort.fn(arr);
            const time = performance.now() - start;
            console.log(`${sort.name}: ${time.toFixed(3)} ms`);
        }
    }
}

compareAllSorts();
