### 题目 5：对比冒泡排序和选择排序（25 分）

**任务：**
编写代码对比两种排序算法的性能。

```javascript
/**
 * 性能对比函数
 */
function compareSorts() {
    const testCases = [
        [64, 34, 25, 12, 22, 11, 90],
        [1, 2, 3, 4, 5, 6, 7],  // 已经有序
        [7, 6, 5, 4, 3, 2, 1],  // 完全逆序
        [5, 5, 5, 5, 5],         // 全部相同
        [3]                       // 单个元素
    ];
    
    for (const arr of testCases) {
        console.log('\n测试数据:', arr);
        
        // 测试冒泡排序
        const start1 = performance.now();
        bubbleSort([...arr]);
        const time1 = performance.now() - start1;
        console.log('冒泡排序耗时:', time1.toFixed(2), 'ms');
        
        // 测试选择排序
        const start2 = performance.now();
        selectionSort([...arr]);
        const time2 = performance.now() - start2;
        console.log('选择排序耗时:', time2.toFixed(2), 'ms');
        
        console.log('速度比:', (time1 / time2).toFixed(2));
    }
}

// 辅助函数
function bubbleSort(arr) {
    const n = arr.length;
    for (let i = 0; i < n - 1; i++) {
        for (let j = 0; j < n - i - 1; j++) {
            if (arr[j] > arr[j + 1]) {
                [arr[j], arr[j + 1]] = [arr[j + 1], arr[j]];
            }
        }
    }
    return arr;
}

compareSorts();
```

### 常见错误提醒

❌ 内层循环边界错误 → ✅ 应该是 j = i + 1
❌ 忘记检查 minIndex !== i → ✅ 避免不必要的交换
