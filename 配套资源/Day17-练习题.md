### 题目 5：对比三种简单排序（25 分）

**任务：**
编写代码对比冒泡、选择和插入排序的性能。

```javascript
/**
 * 性能对比函数
 */
function compareThreeSorts() {
    const testCases = [
        [64, 34, 25, 12, 22, 11, 90],
        [1, 2, 3, 4, 5, 6, 7],  // 已经有序
        [7, 6, 5, 4, 3, 2, 1],  // 完全逆序
        [5, 5, 5, 5, 5],         // 全部相同
        [3]                       // 单个元素
    ];
    
    for (const arr of testCases) {
        console.log('\n测试数据:', arr);
        
        // 测试冒泡排序
        const start1 = performance.now();
        bubbleSort([...arr]);
        const time1 = performance.now() - start1;
        console.log('冒泡排序耗时:', time1.toFixed(2), 'ms');
        
        // 测试选择排序
        const start2 = performance.now();
        selectionSort([...arr]);
        const time2 = performance.now() - start2;
        console.log('选择排序耗时:', time2.toFixed(2), 'ms');
        
        // 测试插入排序
        const start3 = performance.now();
        insertionSort([...arr]);
        const time3 = performance.now() - start3;
        console.log('插入排序耗时:', time3.toFixed(2), 'ms');
        
        // 找出最快的
        const minTime = Math.min(time1, time2, time3);
        if (minTime === time1) console.log('最快：冒泡排序');
        else if (minTime === time2) console.log('最快：选择排序');
        else console.log('最快：插入排序');
    }
}

// 辅助函数
function bubbleSort(arr) {
    const n = arr.length;
    for (let i = 0; i < n - 1; i++) {
        for (let j = 0; j < n - i - 1; j++) {
            if (arr[j] > arr[j + 1]) {
                [arr[j], arr[j + 1]] = [arr[j + 1], arr[j]];
            }
        }
    }
    return arr;
}

function selectionSort(arr) {
    const n = arr.length;
    for (let i = 0; i < n - 1; i++) {
        let minIndex = i;
        for (let j = i + 1; j < n; j++) {
            if (arr[j] < arr[minIndex]) {
                minIndex = j;
            }
        }
        if (minIndex !== i) {
            [arr[i], arr[minIndex]] = [arr[minIndex], arr[i]];
        }
    }
    return arr;
}

function insertionSort(arr) {
    const n = arr.length;
    for (let i = 1; i < n; i++) {
        const current = arr[i];
        let j = i - 1;
        while (j >= 0 && arr[j] > current) {
            arr[j + 1] = arr[j];
            j--;
        }
        arr[j + 1] = current;
    }
    return arr;
}

compareThreeSorts();
```

### 常见错误提醒

❌ 外层循环从 0 开始 → ✅ 应该从 1 开始（第一个元素默认有序）
❌ while 循环条件错误 → ✅ j >= 0 && arr[j] > current
❌ 忘记保存当前元素 → ✅ 先把 arr[i] 保存到 current
