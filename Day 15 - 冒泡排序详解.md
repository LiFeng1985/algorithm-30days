### 💡 排序的定义

**官方说法：**
> 排序是将一组数据按照某种规则（大小、字母顺序等）重新排列成有序序列的过程

**人话版：**
> **排序 = 把乱七八糟的东西按规矩摆好**

```javascript
// 排序的例子

const numbers = [64, 34, 25, 12, 22, 11, 90];

// 升序排序（从小到大）
numbers.sort((a, b) => a - b);
console.log(numbers);
// [11, 12, 22, 25, 34, 64, 90]

// 降序排序（从大到小）
numbers.sort((a, b) => b - a);
console.log(numbers);
// [90, 64, 34, 25, 22, 12, 11]

// 字符串排序
const names = ['小明', '小红', '小刚', '小丽'];
names.sort();
console.log(names);
// 按拼音排序...
```

## 🔵 第 4 步：冒泡排序详解（40 分钟）

### 算法思想

```javascript
/**
 * 冒泡排序（Bubble Sort）
 * 
 * 名字由来：像水中的气泡一样，越浮越高
 * 
 * 核心思想：
 * 1. 比较相邻的两个元素
 * 2. 如果前一个比后一个大，就交换
 * 3. 这样最大的元素会"冒泡"到最后
 * 4. 重复这个过程
 */

// 形象比喻：

/**
   想象一群小朋友按身高排队：
   
   第一轮：
   - 第 1 个和第 2 个比，高的站后面
   - 第 2 个和第 3 个比，高的站后面
   - ...
   - 最后，最高的一定在最后面
   
   第二轮：
   - 同样比，但最后一个不用比了
   - 第二高的在倒数第二
   
   继续...
   
   直到所有人都排好！
*/
```

### 代码实现（基础版）

```javascript
/**
 * 冒泡排序 - 基础版本
 */

function bubbleSort(arr) {
    const n = arr.length;
    
    console.log('原始数组:', arr);
    console.log(`开始排序，共${n}个元素\n`);
    
    // 外层循环：控制轮数
    for (let i = 0; i < n - 1; i++) {
        console.log(`\n=== 第 ${i + 1} 轮 ===`);
        
        // 内层循环：每轮的比较
        for (let j = 0; j < n - 1 - i; j++) {
            console.log(`  比较位置${j}和${j+1}: ${arr[j]} vs ${arr[j+1]}`);
            
            // 如果前面的比后面的大，交换
            if (arr[j] > arr[j + 1]) {
                // 交换
                [arr[j], arr[j + 1]] = [arr[j + 1], arr[j]];
                console.log(`    交换后：${arr[j]} < ${arr[j+1]} ✓`);
            } else {
                console.log(`    不交换：${arr[j]} ≤ ${arr[j+1]} ✓`);
            }
        }
        
        console.log(`第${i + 1}轮结果：`, arr);
    }
    
    console.log('\n排序完成:', arr);
    return arr;
}
```

## ⚡ 第 5 步：优化冒泡排序（20 分钟）

### 优化 1：提前结束

```javascript
/**
 * 问题：如果数组已经有序，还要继续比较吗？
 * 
 * 比如：[1, 2, 3, 4, 5]
 * 第 1 轮发现没有交换，说明已经有序
 * 可以直接结束！
 */

function bubbleSortOptimized(arr) {
    const n = arr.length;
    
    console.log('原始数组:', arr);
    
    for (let i = 0; i < n - 1; i++) {
        let swapped = false;  // 标记是否发生交换
        
        console.log(`\n=== 第 ${i + 1} 轮 ===`);
        
        for (let j = 0; j < n - 1 - i; j++) {
            if (arr[j] > arr[j + 1]) {
                [arr[j], arr[j + 1]] = [arr[j + 1], arr[j]];
                swapped = true;
                console.log(`  交换：${arr[j]} < ${arr[j+1]}`);
            }
        }
        
        // 如果没有交换，说明已经有序
        if (!swapped) {
            console.log('✨ 本轮没有交换，数组已有序，提前结束！');
            break;
        }
        
        console.log(`第${i + 1}轮结果：`, arr);
    }
    
    return arr;
}

// 测试
console.log('\n=== 测试优化版：基本有序的数组 ===');
bubbleSortOptimized([1, 2, 3, 5, 4]);

/*
输出：

原始数组：[1, 2, 3, 5, 4]

=== 第 1 轮 ===
  交换：4 < 5
第 1 轮结果：[1, 2, 3, 4, 5]

=== 第 2 轮 ===
✨ 本轮没有交换，数组已有序，提前结束！

只用了 2 轮就完成！✅
*/
```

### 优化 2：记录最后交换位置

```javascript
/**
 * 进一步优化：
 * 最后一次交换的位置后面，都是有序的
 * 下一轮可以少比较一些
 */

function bubbleSortOptimized2(arr) {
    const n = arr.length;
    let newN = n;  // 记录最后交换位置
    
    console.log('原始数组:', arr);
    
    while (newN > 1) {
        let lastSwap = 0;  // 记录最后交换位置
        
        console.log(`\n=== 新一轮（边界：${newN}）===`);
        
        for (let j = 0; j < newN - 1; j++) {
            if (arr[j] > arr[j + 1]) {
                [arr[j], arr[j + 1]] = [arr[j + 1], arr[j]];
                lastSwap = j + 1;
                console.log(`  交换位置${j}和${j+1}`);
            }
        }
        
        newN = lastSwap;  // 更新边界
        
        if (lastSwap === 0) {
            console.log('✨ 没有交换，结束！');
            break;
        }
        
        console.log('当前数组：', arr);
    }
    
    return arr;
}

// 测试
console.log('\n=== 测试二次优化版 ===');
bubbleSortOptimized2([3, 1, 2, 5, 4]);
```

### 项目：学生成绩排名系统

```javascript
/**
 * 班级成绩管理
 * 
 * 功能：
 * 1. 录入成绩
 * 2. 按总分排序（冒泡排序）
 * 3. 显示排名
 * 4. 统计分数段
 */

class Student {
    constructor(name, chinese, math, english) {
        this.name = name;
        this.chinese = chinese;
        this.math = math;
        this.english = english;
        this.total = chinese + math + english;
        this.average = this.total / 3;
        this.rank = 0;
    }
    
    toString() {
        return `${this.name}(语${this.chinese}/数${this.math}/英${this.english})`;
    }
}

class GradeManager {
    constructor(className) {
        this.className = className;
        this.students = [];
    }
    
    // 添加学生
    addStudent(name, chinese, math, english) {
        const student = new Student(name, chinese, math, english);
        this.students.push(student);
        console.log(`✅ 添加：${student.toString()}，总分${student.total}`);
    }
    
    // 按总分排序（冒泡排序）
    sortByTotal() {
        console.log('\n📊 开始按总分排序...\n');
        
        const arr = this.students;
        const n = arr.length;
        
        for (let i = 0; i < n - 1; i++) {
            let swapped = false;
            
            for (let j = 0; j < n - 1 - i; j++) {
                if (arr[j].total < arr[j + 1].total) {
                    // 交换（降序）
                    [arr[j], arr[j + 1]] = [arr[j + 1], arr[j]];
                    swapped = true;
                }
            }
            
            if (!swapped) break;
        }
        
        // 设置排名
        this.updateRanks();
        
        console.log('✅ 排序完成！\n');
    }
    
    // 更新排名
    updateRanks() {
        let rank = 1;
        this.students[0].rank = rank;
        
        for (let i = 1; i < this.students.length; i++) {
            if (this.students[i].total < this.students[i - 1].total) {
                rank = i + 1;
            }
            this.students[i].rank = rank;
        }
    }
    
    // 显示排名
    showRankings() {
        console.log('\n╔════════════════════════════════════════════╗');
        console.log(`║   ${this.className} 成绩排名                 ║`);
        console.log('╚════════════════════════════════════════════╝\n');
        
        console.log('排名\t姓名\t总分\t平均分\t各科成绩');
        console.log('─'.repeat(70));
        
        for (let s of this.students) {
            const medal = s.rank <= 3 ? ['🥇', '🥈', '🥉'][s.rank - 1] : '  ';
            console.log(`${medal} ${s.rank}\t${s.name}\t${s.total}\t${s.average.toFixed(1)}\t语${s.chinese}/数${s.math}/英${s.english}`);
        }
        
        console.log();
    }
    
    // 统计分数段
    countByRange() {
        console.log('\n📈 分数段统计：\n');
        
        const ranges = [
            { name: '优秀（≥270）', min: 270, max: Infinity, count: 0 },
            { name: '良好（240-269）', min: 240, max: 269, count: 0 },
            { name: '中等（210-239）', min: 210, max: 239, count: 0 },
            { name: '及格（180-209）', min: 180, max: 209, count: 0 },
            { name: '不及格（<180）', min: 0, max: 179, count: 0 }
        ];
        
        for (let s of this.students) {
            for (let r of ranges) {
                if (s.total >= r.min && s.total <= r.max) {
                    r.count++;
                    break;
                }
            }
        }
        
        ranges.forEach(r => {
            const bar = '█'.repeat(r.count);
            console.log(`${r.name.padEnd(12)} ${bar} (${r.count}人)`);
        });
        
        console.log();
    }
    
    // 显示所有学生
    showAll() {
        console.log(`\n${this.className} 共${this.students.length}名学生\n`);
    }
}

// ==================== 测试 ====================

console.log('╔═══════════════════════════════════════╗');
console.log('║   学生成绩排名系统                   ║');
console.log('╚═══════════════════════════════════════╝\n');

const manager = new GradeManager('三年二班');
```
